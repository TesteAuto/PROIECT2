TABEL utilizator

CREATE TABLE `disertatie`.`utilizator` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `utilizator` VARCHAR(255) NOT NULL,
  `parola` VARCHAR(255) NOT NULL,

  `poza` LONGBLOB NOT NULL,

TABEL COMENTARII
CREATE TABLE `disertatie`.`comentarii` (
  `id` INT(100) NOT NULL AUTO_INCREMENT,
  `idUtilizator` INT(50) NOT NULL,
  `titlu` VARCHAR(100) NOT NULL,
  `comentariu` VARCHAR(256) NOT NULL,
  `data` DATE NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `idUtilizator_idx` (`idUtilizator` ASC),
  CONSTRAINT `idUtilizator`
    FOREIGN KEY (`idUtilizator`)
    REFERENCES `disertatie`.`utilizator` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
